# Working example with sea lion genome

options(ucscChromosomeNames=FALSE)
getOption("Gviz.scheme")
## [1] "default"
plot_start <- 1
plot_end <- 3986568 

chr <- "CM019820.2"
# annochr <- "NC_045613.1"

genometrack <- GenomeAxisTrack()

gen <- readDNAStringSet("/home/yakupova/1PROJECTS/Y/assembly/Y.fa")
# sequenceTrack <- SequenceTrack(gen, genome = "ZalCal", chromosome = chr, name = "Sequence")

snp_data <- data.frame(
  chrom = c("CM019820.2", "CM019820.2", "CM019820.2"),
  start = c(50000, 100000, 150000),
  end = c(50000, 100000, 150000),
  name = c("snp1", "snp2", "snp3"),
  score = c(1, 1, 1)
)
snpTrack <- AnnotationTrack(
  start = snp_data$start,
  end = snp_data$end,
  chromosome = snp_data$chrom,
  genome = gen,
  id = snp_data$name,
  feature = snp_data$name,
  name = "SNPs",
  stacking = "dense"
)

# anno <- import("/home/yakupova/1PROJECTS/Y/ncbi_annotation/Ygenes2020-08.gff3")
# anno <- import("/home/yakupova/1PROJECTS/Y/ncbi_annotation/annotation_CM019821.2.gff3")
# geneTrack <- GeneRegionTrack(anno, genome = "ZalCal", chromosome = chr, name = "Genes")
Yregion_colors <- c(
  "pseudo" = "mediumseagreen",
  "single" = "steelblue3",
  "multi" = "khaki1",
  "Xtrans" = "palevioletred2"
)

annotation_file <- "/home/yakupova/1PROJECTS/Y/Y_annotation_plot/Ychromsome_annotation_Rin.csv"
annotations <- read.csv(annotation_file, sep = '\t')
geneTrack <- AnnotationTrack(
                genome = "ZalCal", chromosome = chr, name = "Genes",
                start = annotations$Start,
                end = annotations$Stop,
                strand = annotations$Strand,
                group = annotations$Name,
                feature = annotations$Name,
                fill = Yregion_colors[annotations$Yregion],
                col = Yregion_colors[annotations$Yregion],
                stacking="dense", showFeatureNames=TRUE,
                fontcolor.group = "black", fontcolor.item = "black",
                shape="fixedArrow", arrowHeadWidth=100, showFeatureId=TRUE,
                just.group = "above", featureAnnotation = "above",
                rotation.item = 90, height = 0.5)

bam <- "/home/yakupova/1PROJECTS/Y/bam/G465_merged_duprm_subsample0_25_realigned.Y.bam"
coverageTrack <- DataTrack(range = bam, genome = "ZalCal", type = "l",
                           name = "Coverage", window = -1,
                           chromosome = chr, ylim = c(0, 20))


plotTracks(list(genometrack, snpTrack, geneTrack, coverageTrack),
           from = plot_start, to = plot_end,
           chromosome = chr,
           background.panel = "white", background.title = "white", col.axis = "black", col.title = "black")
